# ملاحظة: هذا الهيكل يدعم 9 مراحل (2 السابقة + 7 الجديدة) class GameEngine: def __init__(self): self.current_stage = 1 self.total_stages = 9 # تم التحديث إلى 9 مراحل self.score = 0 def load_stage(self, stage_num): print(f"--- تحميل المرحلة رقم {stage_num} ---") # هنا يتم وضع تفاصيل كل مرحلة stages_data = { 1: "مرحلة البداية: تدريب بسيط", 2: "المرحلة الثانية: زيادة السرعة", 3: "المرحلة الثالثة: ظهور العوائق الثابتة", 4: "المرحلة الرابعة: العوائق المتحركة", 5: "المرحلة الخامسة: تحدي الوقت (سرعة مضاعفة)", 6: "المرحلة السادسة: الجاذبية المتغيرة", 7: "المرحلة السابعة: المتاهة المعقدة", 8: "المرحلة الثامنة: هجوم الأعداء", 9: "مرحلة الزعيم: التحدي النهائي لهذه المجموعة" } return stages_data.get(stage_num, "مرحلة غير معروفة") def next_stage(self): if self.current_stage < self.total_stages: self.current_stage += 1 print(f"مبروك! انتقلت إلى المرحلة {self.current_stage}") else: print("تهانينا! أتممت جميع المراحل السبعة الجديدة بنجاح!") # تشغيل التجربة game = GameEngine() for i in range(9): print(game.load_stage(game.current_stage)) game.next_stage()
